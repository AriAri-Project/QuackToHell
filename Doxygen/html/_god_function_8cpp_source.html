60FString UGodFunction::ExtractValidJson(FString AIResponse)
61{
62 // OpenAI ì‘ë‹µì„ JSON ê°ì²´ë¡œ íŒŒì‹±
63 TSharedPtr<FJsonObject> RootObject;
64 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(AIResponse);
65
66 if (FJsonSerializer::Deserialize(Reader, RootObject) && RootObject->HasField("choices"))
67 {
68 TArray<TSharedPtr<FJsonValue>> Choices = RootObject->GetArrayField("choices");
69
70 if (Choices.Num() > 0 && Choices[0]->AsObject()->HasField("message"))
71 {
72 FString JsonContent = Choices[0]->AsObject()->GetObjectField("message")->GetStringField("content");
73
74 // Markdown ì½”ë“œ ë¸”ë¡(```json ... ```) ì œê±°
75 JsonContent.ReplaceInline(TEXT("```json"), TEXT(""));
76 JsonContent.ReplaceInline(TEXT("```"), TEXT(""));
77 JsonContent = JsonContent.TrimStartAndEnd();
78
79 UE_LOG(LogTemp, Log, TEXT("Extracted JSON Content: %s"), *JsonContent);
80 return JsonContent;
81 }
82 }
83
84 UE_LOG(LogTemp, Error, TEXT("Failed to extract valid JSON from AI response!"));
85 return TEXT("");

89{
90 TSharedPtr<FJsonObject> JsonObject;

92
93 if (FJsonSerializer::Deserialize(Reader, JsonObject))
94 {
95 if (JsonObject->HasField("name"))

98
99 // JSON ë‚´ì— ê°™ì€ ì´ë¦„ì´ ì—¬ëŸ¬ ë²ˆ ë“¤ì–´ê°€ë©´ ì œê±°
100 JsonObject->SetStringField("name", Name);
101 }
102
103 FString CleanJson;
104 TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&CleanJson);
105 FJsonSerializer::Serialize(JsonObject.ToSharedRef(), Writer);
106
107 UE_LOG(LogTemp, Log, TEXT("Cleaned JSON: %s"), *CleanJson);
108 return CleanJson;
109 }
110
111 UE_LOG(LogTemp, Error, TEXT("Failed to clean JSON!"));
