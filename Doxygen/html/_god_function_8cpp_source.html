46// JSON Escape 함수 추가 (OpenAI API 관련 함수들 위쪽)
47FString UGodFunction::EscapeJSON(const FString& Input)
48{
49 FString Escaped = Input;
50 Escaped.ReplaceInline(TEXT("\\"), TEXT("\\\\"));
51 Escaped.ReplaceInline(TEXT("\""), TEXT("\\\""));
52 Escaped.ReplaceInline(TEXT("\n"), TEXT("\\n"));
53 Escaped.ReplaceInline(TEXT("\r"), TEXT(""));
54 Escaped.ReplaceInline(TEXT("\t"), TEXT("\\t"));
55
56 return Escaped;
57}

60FString UGodFunction::ExtractValidJson(FString AIResponse)
61{
62 // OpenAI 응답을 JSON 객체로 파싱
63 TSharedPtr<FJsonObject> RootObject;
64 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(AIResponse);
65
66 if (FJsonSerializer::Deserialize(Reader, RootObject) && RootObject->HasField("choices"))
67 {
68 TArray<TSharedPtr<FJsonValue>> Choices = RootObject->GetArrayField("choices");
69
70 if (Choices.Num() > 0 && Choices[0]->AsObject()->HasField("message"))
71 {
72 FString JsonContent = Choices[0]->AsObject()->GetObjectField("message")->GetStringField("content");
73
74 // Markdown 코드 블록(```json ... ```) 제거
75 JsonContent.ReplaceInline(TEXT("```json"), TEXT(""));
76 JsonContent.ReplaceInline(TEXT("```"), TEXT(""));
77 JsonContent = JsonContent.TrimStartAndEnd();
78
79 UE_LOG(LogTemp, Log, TEXT("Extracted JSON Content: %s"), *JsonContent);
80 return JsonContent;
81 }
82 }
83
84 UE_LOG(LogTemp, Error, TEXT("Failed to extract valid JSON from AI response!"));
85 return TEXT("");
86}

88FString UGodFunction::CleanUpJson(FString JsonString)
89{
90 TSharedPtr<FJsonObject> JsonObject;
91 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(JsonString);
92
93 if (FJsonSerializer::Deserialize(Reader, JsonObject))
94 {
95 if (JsonObject->HasField("name"))
96 {
97 FString Name = JsonObject->GetStringField("name");
98
99 // JSON 내에 같은 이름이 여러 번 들어가면 제거
100 JsonObject->SetStringField("name", Name);
101 }
102
103 FString CleanJson;
104 TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&CleanJson);
105 FJsonSerializer::Serialize(JsonObject.ToSharedRef(), Writer);
106
107 UE_LOG(LogTemp, Log, TEXT("Cleaned JSON: %s"), *CleanJson);
108 return CleanJson;
109 }
110
111 UE_LOG(LogTemp, Error, TEXT("Failed to clean JSON!"));
112 return JsonString;
113}

168// OpenAI API 호출 함수
169void UGodFunction::CallOpenAIAsync(const FString& Prompt, TFunction<void(FString)> Callback, int RetryCount)
170{
171 FString ApiKey = LoadOpenAIKey();
172 if (ApiKey.IsEmpty())
173 {
174 UE_LOG(LogTemp, Error, TEXT("OpenAI API Key is missing!"));
175 Callback(TEXT(""));
176 return;
177 }
178
179 TSharedPtr<IHttpRequest> Request = FHttpModule::Get().CreateRequest();
180 Request->SetURL("https://api.openai.com/v1/chat/completions");
181 Request->SetVerb("POST");
182 Request->SetHeader("Authorization", "Bearer " + ApiKey);
183 Request->SetHeader("Content-Type", "application/json");
184
185 FString PostData = FString::Printf(
186 TEXT("{ \"model\": \"gpt-4o\", \"messages\": [{ \"role\": \"user\", \"content\": \"%s\" }], \"max_tokens\": 2048 }"),
187 *EscapeJSON(Prompt)
188 );
189
190 Request->SetContentAsString(PostData);
191
192 UE_LOG(LogTemp, Log, TEXT("OpenAI API Request Sent (Attempt %d): %s"), RetryCount + 1, *PostData);
193
194 Request->OnProcessRequestComplete().BindLambda([Callback, Prompt, RetryCount](FHttpRequestPtr Request, FHttpResponsePtr Response, bool bWasSuccessful)
195 {
196 if (bWasSuccessful && Response.IsValid())
197 {
198 FString AIResponse = Response->GetContentAsString();
199 UE_LOG(LogTemp, Log, TEXT("Raw OpenAI Response: %s"), *AIResponse);
200
201 // JSON 추출 및 정리
202 FString CleanedJson = UGodFunction::ExtractValidJson(AIResponse);
203
204 if (!CleanedJson.IsEmpty())
205 {
206 Callback(CleanedJson);
207 }
208 // 최대 3번까지 재시도
209 else if (RetryCount < 2)
210 {
211 UE_LOG(LogTemp, Warning, TEXT("OpenAI 응답이 비어 있음. %d번째 재시도 중..."), RetryCount + 1);
212 FPlatformProcess::Sleep(1.0f);
213 UGodFunction::CallOpenAIAsync(Prompt, Callback, RetryCount + 1);
214 }
215 else
216 {
217 UE_LOG(LogTemp, Error, TEXT("OpenAI 응답이 3회 연속 실패. 요청 중단."));
218 Callback(TEXT(""));
219 }
220 }
221 // 실패 시 최대 3번 재시도
222 else if (RetryCount < 2)
223 {
224 UE_LOG(LogTemp, Warning, TEXT("OpenAI 요청 실패. %d번째 재시도 중..."), RetryCount + 1);
225 FPlatformProcess::Sleep(1.0f);
226 UGodFunction::CallOpenAIAsync(Prompt, Callback, RetryCount + 1);
227 }
228 else
229 {
230 UE_LOG(LogTemp, Error, TEXT("OpenAI 요청이 3회 연속 실패. 요청 중단."));
231 Callback(TEXT(""));
232 }
233 });
234
235 Request->ProcessRequest();
236}

239void UGodFunction::DeleteOldPromptFiles()
240{
241
242 static bool bAlreadyDeleted = false; // ✅ 중복 실행 방지
243 if (bAlreadyDeleted)
244 {
245 UE_LOG(LogTemp, Log, TEXT("🛑 DeleteOldPromptFiles()가 이미 실행되었으므로 재실행 방지"));
246 return;
247 }
248
249 FString PromptFolder = FPaths::ProjectSavedDir() + TEXT("Prompt/");
250
251 if (!FPaths::DirectoryExists(PromptFolder))
252 {
253 UE_LOG(LogTemp, Warning, TEXT("Prompt folder does not exist, skipping deletion."));
254 return;
255 }
256
257 IFileManager& FileManager = IFileManager::Get();
258 // 특정 파일 제외한 파일들만 삭제
259 TArray<FString> Files;
260 FileManager.FindFiles(Files, *PromptFolder, TEXT(".json"));
261
262 for (const FString& File : Files)
263 {
264 if (File.Contains("PromptToGod"))
265 {
266 UE_LOG(LogTemp, Log, TEXT("Skipping essential file: %s"), *File);
267 continue;
268 }
269
270 FString FilePath = PromptFolder + File;
271 FilePath = FPaths::ConvertRelativePathToFull(FilePath);
272
273 if (FileManager.Delete(*FilePath))
274 {
275 UE_LOG(LogTemp, Log, TEXT("Deleted old Prompt file: %s"), *FilePath);
276 }
277 else
278 {
279 UE_LOG(LogTemp, Error, TEXT("Failed to delete file: %s"), *FilePath);
280 }
281 }
282 bAlreadyDeleted = true; // ✅ 삭제가 한 번만 실행되도록 설정
283 if (UQGameInstanceVillage* GameInstance = Cast<UQGameInstanceVillage>(GEngine->GetWorldContextFromGameViewport(GEngine->GameViewport)->World()->GetGameInstance()))
284 {
285 GameInstance->StartPromptGeneration();
286 }
