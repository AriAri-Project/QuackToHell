17AQNPC::AQNPC(const FObjectInitializer& ObjectInitializer)
18 :Super(ObjectInitializer)
19{ /*위치값 세팅*/
20 this->GetMesh()->SetRelativeLocation(FVector(0.f, 0.f, -50.f));
21 this->GetMesh()->SetRelativeRotation(FRotator(0.f, -90.f, 0.f));
22 /*캡슐 콜라이더 세팅*/
23 this->GetCapsuleComponent()->InitCapsuleSize(50.0f, 60.0f);
24 /*NPC 컴포넌트*/
25 NPCComponent = CreateDefaultSubobject<UNPCComponent>(TEXT("NPCComponent"));
26 /*말풍선 UI 컴포넌트*/
27 this->SpeechBubbleWidgetComponent = CreateDefaultSubobject<UWidgetComponent>(TEXT("SpeechBubbleWidget"));
28 EWidgetSpace WidgetSpace = EWidgetSpace::Screen;
29 this->SpeechBubbleWidgetComponent->SetWidgetSpace(WidgetSpace);
30 this->SpeechBubbleWidgetComponent->SetDrawAtDesiredSize(true);
31 this->SpeechBubbleWidgetComponent->SetupAttachment(RootComponent);
32 // **위젯 클래스는 블루프린트 에디터에서**:
33 // 해당 컴포넌트의 Details → Widget Class 에 WBP_QSpeechBubble을 드래그해서 지정.
34 // 언리얼이 클라이언트에서 알아서 BP 위젯을 생성해 줍니다.
35
36 /*충돌처리*/
37 InteractionSphereComponent = CreateDefaultSubobject<USphereComponent>(TEXT("InteractionSphere"));
38 InteractionSphereComponent->SetupAttachment(RootComponent);
39 InteractionSphereComponent->SetSphereRadius(SphereRadius);
40
41 /*충돌처리 바인딩*/
42 InteractionSphereComponent->OnComponentBeginOverlap.AddDynamic(this, &AQNPC::OnOverlapBegin);
43 InteractionSphereComponent->OnComponentEndOverlap.AddDynamic(this, &AQNPC::OnOverlapEnd);
44}
