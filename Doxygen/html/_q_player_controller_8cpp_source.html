96void AQPlayerController::SetupInputComponent()
97{
98 Super::SetupInputComponent();
99
100 //InputComponent는 Actor의 멤버변수이다. 멤버변수를 EnhancedInput으로 캐스팅한다.
101 auto* EnhancedInputComponent = Cast<UEnhancedInputComponent>(InputComponent);
102 if (EnhancedInputComponent) {
103 /*액션에 대해, 트리거 되면 콜백되는 함수를 바인딩한다.*/
104 EnhancedInputComponent->BindAction(MoveAction, ETriggerEvent::Triggered, this, &ThisClass::InputMove);
105 EnhancedInputComponent->BindAction(TurnAction, ETriggerEvent::Triggered, this, &ThisClass::InputTurn);
106 EnhancedInputComponent->BindAction(InteractionAction, ETriggerEvent::Started, this, &ThisClass::InputInteraction);
107 EnhancedInputComponent->BindAction(EnableTurnAction, ETriggerEvent::Started, this, &ThisClass::InputEnableTurn);
108 EnhancedInputComponent->BindAction(TurnOnOffMapAction, ETriggerEvent::Started, this, &ThisClass::InputTurnOnOffMap);
109 EnhancedInputComponent->BindAction(TurnOnOffInventoryAction, ETriggerEvent::Started, this, &ThisClass::InputTurnOnOffInventory);
110 EnhancedInputComponent->BindAction(EnableTurnAction, ETriggerEvent::Completed, this, &ThisClass::InputEnableTurn);
111 }

227void AQPlayerController::FreezePawn()
228{
229 //폰 정보 가져오기
230 TObjectPtr<APawn> ControlledPawn = this->GetPawn();
231 if (!ControlledPawn) {
232 return;
233 }
234
235 //movement component 가져오기
236 TObjectPtr<UCharacterMovementComponent> MovementComponent = ControlledPawn->FindComponentByClass<UCharacterMovementComponent>();
237 if (MovementComponent) {
238 MovementComponent->StopMovementImmediately();//즉시멈춤(속도를 0으로 만듦)
239 MovementComponent->SetComponentTickEnabled(false);//틱 비활성화
240 MovementComponent->Deactivate();//이동 비활성화
241 }
242
243 UE_LOG(LogLogic, Log, TEXT("플레이어 멈춤."));

350void AQPlayerController::InputMove(const FInputActionValue& InputValue)
351{
352
353 FVector2D MovementVector = InputValue.Get<FVector2D>();
354
355 //앞 이동
356 if (MovementVector.X != 0) {
357 /*내가 바라보는 곳 기준의 앞 방향을 구해줌*/
358 //마우스 방향 회전값 기준
359 FRotator Rotator = this->GetControlRotation();
360 //해당 회전값의 기준의 앞 방향을 구해준다.
361 FVector Direction = UKismetMathLibrary::GetForwardVector(FRotator(0, Rotator.Yaw, 0));
362 //방향과 크기를 전달함. (시간마다 거리를 더하는 것은 MovementComponent에서, 속도 세팅은 movementcomponent에서)
363 this->GetPawn()->AddMovementInput(Direction, MovementVector.X);
364 }
365 //옆 이동
366 if (MovementVector.Y != 0) {
367 /*내가 바라보는 곳 기준의 옆 방향을 구해줌*/
368 //마우스 방향 회전값 기준
369 FRotator Rotator = this->GetControlRotation();
370 //해당 회전값의 기준의 옆 방향을 구해준다.
371 FVector Direction = UKismetMathLibrary::GetRightVector(FRotator(0, Rotator.Yaw, 0));
372 //방향과 크기를 전달함. (시간마다 거리를 더하는 것은 MovementComponent에서, 속도 세팅은 movementcomponent에서)
373 this->GetPawn()->AddMovementInput(Direction, MovementVector.Y);
374 }
