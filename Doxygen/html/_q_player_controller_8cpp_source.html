64void AQPlayerController::InputTurn(const FInputActionValue& InputValue)
65{
66 //bEnableTurn이 true일때만 화면전환이 가능하다.
67 if (bEnableTurn) {
68 UE_LOG(LogLogic, Log, TEXT("%f, %f"), InputValue.Get<FVector2D>().X, (InputValue.Get<FVector2D>().Y*-1));
69 //내 고개 회전값을 control rotation에 전달. : 상하좌우 고갯짓
70 float Yaw = InputValue.Get<FVector2D>().X;
71 float Pitch = InputValue.Get<FVector2D>().Y;
72 this->AddYawInput(Yaw);
73 this->AddPitchInput(-Pitch);
74 }

79void AQPlayerController::InputMove(const FInputActionValue& InputValue)
80{
81 UE_LOG(LogLogic, Log, TEXT("%f, %f"), InputValue.Get<FVector2D>().X, InputValue.Get<FVector2D>().Y);
82
83 FVector2D MovementVector = InputValue.Get<FVector2D>();
84
85 //앞 이동
86 if (MovementVector.X != 0) {
87 /*내가 바라보는 곳 기준의 앞 방향을 구해줌*/
88 //마우스 방향 회전값 기준
89 FRotator Rotator = this->GetControlRotation();
90 //해당 회전값의 기준의 앞 방향을 구해준다.
91 FVector Direction = UKismetMathLibrary::GetForwardVector(FRotator(0, Rotator.Yaw, 0));
92 //방향과 크기를 전달함. (시간마다 거리를 더하는 것은 MovementComponent에서, 속도 세팅은 movementcomponent에서)
93 this->GetPawn()->AddMovementInput(Direction, MovementVector.X);
94 }
95 //옆 이동
96 if (MovementVector.Y != 0) {
97 /*내가 바라보는 곳 기준의 옆 방향을 구해줌*/
98 //마우스 방향 회전값 기준
99 FRotator Rotator = this->GetControlRotation();
100 //해당 회전값의 기준의 옆 방향을 구해준다.
101 FVector Direction = UKismetMathLibrary::GetRightVector(FRotator(0, Rotator.Yaw, 0));
102 //방향과 크기를 전달함. (시간마다 거리를 더하는 것은 MovementComponent에서, 속도 세팅은 movementcomponent에서)
103 this->GetPawn()->AddMovementInput(Direction, MovementVector.Y);
104 }
105}
