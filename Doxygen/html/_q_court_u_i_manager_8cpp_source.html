144void AQCourtUIManager::TurnOnUI(ECourtUIType UIType)
145{
146 /*예외처리*/
147 //키에 매핑되는 클래스목록이 존재하지 않으면
148 if (!UIWidgetClasses.Contains(UIType)) {
149 UE_LOG(LogLogic, Warning, TEXT("QCourtUIManager: 키에 매핑되는 클래스목록이 존재하지 않습니다."));
150 return;
151 }
152
153 /*UI켜기*/
154 //위젯 이미 만들어져있으면 visible 전환
155 if (ActivedCourtWidgets.Contains(UIType)) {
156 ActivedCourtWidgets[UIType]->SetVisibility(ESlateVisibility::Visible);
157 return;
158 }
159
160 //없으면 위젯 새로 만들기 & visible처리
161 TObjectPtr<UWorld> World = GetWorld();
162 if (!World) return;
163
164 TObjectPtr<UUserWidget> NewWidget = CreateWidget<UUserWidget>(World, UIWidgetClasses[UIType]);
165 if (NewWidget) {
166 NewWidget->AddToViewport();
167 NewWidget->SetVisibility(ESlateVisibility::Visible);
168 ActivedCourtWidgets.Add(UIType, NewWidget);
169 }
170}

180TObjectPtr<AQCourtUIManager> AQCourtUIManager::GetInstance(TObjectPtr<UWorld> World)
181{
182 if (!Instance)
183 {
184 // ✅ 클라이언트에서도 `AQVillageUIManager` 찾도록 설정
185 for (TActorIterator<AQCourtUIManager> It(World); It; ++It)
186 {
187 Instance = *It;
188 break;
189 }
190
191 // ✅ 그래도 없으면 생성 (서버에서만 실행)
192 if (!Instance && World->GetNetMode() != NM_Client)
193 {
194 Instance = World->SpawnActor<AQCourtUIManager>(AQCourtUIManager::StaticClass());
195 }
196 }
197 return Instance;
198}
