140 const TArray<FConversationRecord> GetRecordWithID(int32 ID) const
141 {
142 // 대화기록이 없다면 바로 return
143 if (ConversationList.Num() == 0)
144 {
145 UE_LOG(LogTemp, Warning, TEXT("ConversationList is empty in GetRecordWithID!"));
146 return TArray<FConversationRecord>();
147 }
148
149 return ConversationList.FilterByPredicate([ID](const FConversationRecord& Record)
150 {
151 // Player와 관련된 대화가 아니라면 제외
152 if (Record.GetConversationType() != EConversationType::P2N && Record.GetConversationType() != EConversationType::PStart)
153 {
154 return false;
155 }
156 return (Record.GetSpeakerID() == ID || Record.GetListenerID() == ID);
157 });
