39TObjectPtr<AActor> AQDynamicNPC::GetClosestNPC()
40{
41 if (OverlappingNPCs.Num() == 0) {
42 //대화 대상 없음
43 return nullptr;
44 }
45
46 TObjectPtr<AActor> ClosestNPC = nullptr;
47 float MinDistance = FLT_MAX;
48
49 for (TObjectPtr<AActor> NPC : OverlappingNPCs) {
50 //캐릭터와 NPC간 거리
51 float Distance = FVector::Dist(this->GetActorLocation(), NPC->GetActorLocation());
52 //최소거리찾기
53 if (Distance < MinDistance) {
54 MinDistance = Distance;
55 ClosestNPC = NPC;
56 }
57 }
58
59 return ClosestNPC;
60}

64bool AQDynamicNPC::GetResponse(AQPlayerController* ClientPC, FString& Text, EConversationType InputConversationType)
65{
66 // 응답 요청을 위한 FOpenAIRequest 구조체 구성
67 FOpenAIRequest OpenAIRequest;
68 OpenAIRequest.ConversationType = InputConversationType;
69 AQPlayerController* PlayerController = Cast<AQPlayerController>(UGameplayStatics::GetPlayerController(this, 0));
70 if (PlayerController == nullptr)
71 {
72 UE_LOG(LogLogic, Error, TEXT("AQDynamicNPC::GetResponse - PlayerController is nullptr."));
73 return false;
74 }
75 OpenAIRequest.ListenerID = Cast<AQPlayerState> (PlayerController->PlayerState)->GetPlayerId();
76 OpenAIRequest.Prompt = Text;
77 OpenAIRequest.SpeakerID = NPCComponent->GetNPCID();
78
79 // 서버에게 AI 응답 요청 보내기
80 NPCComponent->ServerRPCGetNPCResponse(ClientPC, OpenAIRequest);
81 return true;

104 //N2N : 만약 NPC가 다가왔을 시, 대화 시작하기

107 OpponentNPC = Cast<AQDynamicNPC>(GetClosestNPC());
108 if (OpponentNPC) {
109
110 //조건
111 // 내가 대화기록 있으면 : 플레이어와 대화한 기록이 있는 망자NPC가 먼저[망자 NPC – 망자 NPC 대화말풍선 UI]를 나타낸다
112 AQPlayerState* _PlayerState = GetWorld()->GetFirstPlayerController()->GetPlayerState<AQPlayerState>();

114 //플레이어스테이트로 대화기록 조회
115 int32 NPCID = NPCComponent->GetNPCID();

119 }
120 else {
121 UE_LOG(LogLogic, Error, TEXT("AQDynamicNPCController::StartDialog - PlayerState is nullptr"));
122 }
123
124 //대화기록이 있는지없는지 확인
125 if (!ConversationRecords.IsEmpty()) {
126 //대화기록 있는 애이니, 먼저 말 건다.
127 AQDynamicNPCController* MyController = Cast<AQDynamicNPCController>(GetController());
128 if (MyController == nullptr)
129 {
130 UE_LOG(LogLogic, Error, TEXT("AQDynamicNPC::OnOverlapBegin - Controller is nullptr."));
131 return;
132 }
133 else {
134 MyController->StartDialog(OpponentNPC, ENPCConversationType::N2N);
135 }
136 }
