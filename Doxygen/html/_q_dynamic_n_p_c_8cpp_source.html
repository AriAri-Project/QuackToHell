34TObjectPtr<AActor> AQDynamicNPC::GetClosestNPC()
35{
36 if (OverlappingNPCs.Num() == 0) {
37 //대화 대상 없음
38 return nullptr;
39 }
40
41 TObjectPtr<AActor> ClosestNPC = nullptr;
42 float MinDistance = FLT_MAX;
43
44 for (TObjectPtr<AActor> NPC : OverlappingNPCs) {
45 //캐릭터와 NPC간 거리
46 float Distance = FVector::Dist(this->GetActorLocation(), NPC->GetActorLocation());
47 //최소거리찾기
48 if (Distance < MinDistance) {
49 MinDistance = Distance;
50 ClosestNPC = NPC;
51 }
52 }
53
54 return ClosestNPC;
55}
