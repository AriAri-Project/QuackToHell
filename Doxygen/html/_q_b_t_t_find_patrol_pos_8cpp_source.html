18EBTNodeResult::Type UQBTTFindPatrolPos::ExecuteTask(UBehaviorTreeComponent& OwnerComp, uint8* NodeMemory)
19{
20 /*AI Controller 가져오기*/
21 TObjectPtr<AAIController> AIController = OwnerComp.GetAIOwner();
22 if (!AIController) {
23 return EBTNodeResult::Failed;
24 }
25
26 /*캐릭터 가져오기*/
27 TObjectPtr<ACharacter> ControlledCharacter = Cast<ACharacter>(AIController->GetPawn());
28 if (!ControlledCharacter) {
29 return EBTNodeResult::Failed;
30 }
31
32 /*blackboard 가져오기*/
33 const TObjectPtr<UBlackboardComponent> BlackboardComp = AIController->GetBlackboardComponent();
34 if (!BlackboardComp) {
35 return EBTNodeResult::Failed;
36 }
37
38 /*원형범위 내에서 랜덤한 patrol 위치 선택*/
39 const FVector CurrentLocation = ControlledCharacter->GetActorLocation();
40 //순찰반경 반지름
41 PatrolRadius = 500.0f;
42 //꼭짓점에서 어떤 각도로 갈건지
43 const float RandomAngle = FMath::RandRange(0.f, 360.f);
44 //라디안으로 변환: FMath::Cos(). FMath::Sin()이 라디안단위를 사용하므로
45 const float RadAngle = FMath::DegreesToRadians(RandomAngle);
46 //꼭짓점에서 얼마만큼 떨어진 곳으로 갈건지
47 const float RandomDistance = FMath::RandRange(MinimumDistance, PatrolRadius);
48
49 /*patrol 위치계산: 방향(각도) * 거리(반지름) = 위치 벡터 */
50 const FVector PatrolOffset = FVector
51 (FMath::Cos(RadAngle) * RandomDistance, //x
52 FMath::Sin(RadAngle) * RandomDistance, //y
53 0.0f); // z: 평면 이동이므로 Z축 값은 0으로 고정
54 //pawn기준 얼마나 떨어져있는지.
55 const FVector NewPatrolLocation = CurrentLocation + PatrolOffset;
56
57 /*블랙보드에 새 patrolpos 설정*/
58 BlackboardComp->SetValueAsVector(PatrolPosKey, NewPatrolLocation);
59
60 return EBTNodeResult::Succeeded;
61}
