78void UQRecordWidget::UpdateRecordHistory()
79{
80 /*1. PlayerState에서 새로 들어온 대화 레코드 정보가져오기*/
81 APlayerController* PlayerController = GetOwningPlayer();
82 AQPlayerState* PlayerState=nullptr;
83 if (PlayerController) {
84 PlayerState = PlayerController->GetPlayerState<AQPlayerState>();
85 if (PlayerState) {
86 ConversationRecord = PlayerState->GetRecordWithPlayerID();
87 }
88 }
89
90 if (ConversationRecord.Num() == 0)
91 {
92 UE_LOG(LogTemp, Log, TEXT("UQRecordWidget UpdateRecordHistory : ConversationRecord is empty"))
93 return;
94 }
95
96 /*대화기록 정보 재구성: NPC별로, 시간순정렬*/
97 //대화한 NPC목록들 저장
98 //2. 기존에 있는 SortedConversationRecord을 덮어쓰기 대신 누적 방식으로 유지
99 for (const FConversationRecord& Record : ConversationRecord) {
100 //누락방지: Speaker나 Listener 둘 중 하나가 NPC면 기록대상
101 int32 PossibleNPCID = -1;
102
103 if (Record.GetSpeakerID() >= 2000) {
104 PossibleNPCID = Record.GetSpeakerID();
105 }
106 else if (Record.GetListenerID() >= 2000) {
107 PossibleNPCID = Record.GetListenerID();
108 }
109 else {
110 continue;//NPC가 안 껴있는대화는 생략
111 }
112
113 //NPC목록에 없으면 추가
114 if (!ConversedNPC.Contains(PossibleNPCID))
115 {
116 ConversedNPC.Add(PossibleNPCID);
117 }
118
119 //NPC에 해당하는 배열이 없으면 새로 생성
120 if (!SortedConversationRecord.Contains(PossibleNPCID))
121 {
122 {
123 SortedConversationRecord.Add(PossibleNPCID, {});
124 }
125 }
126
127 //중복 삽입 방지: 같은 ConversationID가 없을 때만 추가
128 TArray<FConversationRecord>& RecordsForNPC = SortedConversationRecord[PossibleNPCID];
129 bool bAlreadyExists = RecordsForNPC.ContainsByPredicate([&](const FConversationRecord& Existing) {
130 return Existing.GetConversationID() == Record.GetConversationID();
131 });
132 if (!bAlreadyExists){
133 RecordsForNPC.Add(Record);
134 }
135 }
136
137 //3. 모든 NPC기록을 시간순정렬
138 for (auto& Pair : SortedConversationRecord)
139 {
140 Pair.Value.Sort([](const FConversationRecord& A, const FConversationRecord& B) {
141 return A.GetTimestamp() < B.GetTimestamp();
142 });
143 }
144
145 /*위젯업데이트: 버튼 클릭시 콜백*/
146 /*전체 버튼 모양(8개 버튼들) 업데이트: 대화한 NPC 있을경우 유령모양 + NPC이름 뜨게*/
147 UpdateButtonOverlay();
148}

151void UQRecordWidget::TurnOnChatWidget(FText NPCID)
152{
153 /*이전 흔적 파괴: 채팅내용 새롭게 덮어쓰기*/
154 //vertical box의 자식을 삭제한다.
155 VerticalBox->ClearChildren();
156
157 /*버튼에 해당하는 NPC ID 추출 */
158 int32 TargetNPCID = FCString::Atoi(*NPCID.ToString());
159
160 /*SortedConversationRecord에 해당 NPC가 있는지 확인*/
161 if (!SortedConversationRecord.Contains(TargetNPCID))
162 {
163 UE_LOG(LogLogic, Error, TEXT("UQRecordWidget TurnOnChatWidget: 해당 NPCID가 없습니다!"));
164 return;
165 }
166
167 //NPCID키로 매핑하여 대화기록 가져옴
168 const TArray<FConversationRecord>& ChatRecords = SortedConversationRecord[TargetNPCID];
169 TObjectPtr<UWorld> World = GetWorld();
170 if (!World) {
171 return;
172 }
173
174 /*대화 기록 순서대로 위젯 생성*/
175 for (const FConversationRecord& Record : ChatRecords) {
176 //1. 채팅 위젯 생성
177 TObjectPtr<UQChatBoxWidget> NewChatBox = CreateWidget<UQChatBoxWidget>(World, ClassOfChatBox);
178 if (!NewChatBox) {
179 continue;
180 }
181
182 NewChatBox->AddToViewport();
183 NewChatBox->SetVisibility(ESlateVisibility::Visible);
184
185 //2. 색상설정 (NPC: 연두색, Player: 흰색)
186 bool bIsNPCSpeaking = (Record.GetSpeakerID() == TargetNPCID);
187 FLinearColor TextColor = bIsNPCSpeaking
188 ? FLinearColor(0.886f, 1.0f, 0.890f, 1.0f) // 연두색
189 : FLinearColor(1.0f, 1.0f, 1.0f, 1.0f); // 흰색
190
191 NewChatBox->SetColorAndOpacity(TextColor);
192
193 //3. 메세지 설정
194 TObjectPtr<UTextBlock> TextBox = Cast<UTextBlock>(NewChatBox->GetWidgetFromName(TEXT("Text")));
195 if (TextBox) {
196 TextBox->SetText(FText::FromString(Record.GetMessage()));
197 }
198
199 //4. 정렬 방향 설정
200 TObjectPtr<UVerticalBoxSlot> NewSlot = VerticalBox->AddChildToVerticalBox(NewChatBox);
201 if (NewSlot) {
202 EHorizontalAlignment Alignment = bIsNPCSpeaking ? HAlign_Left : HAlign_Right;
203 NewSlot->SetHorizontalAlignment(Alignment);
204
205 NewSlot->SetPadding(FMargin(20.0f, 20.0f, 20.0f, 20.0f));
206 }
207 }
208}
