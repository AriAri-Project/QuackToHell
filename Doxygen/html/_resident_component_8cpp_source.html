9void UResidentComponent::PerformNPCLogic()
10{
11 // 주민 컴포넌트의 로직 시작 로그 출력
12 UE_LOG(LogTemp, Log, TEXT("Resident is providing information about the Defendant's past."));
13
14 // 주민 JSON 파일 경로 정의 (주민은 4명)
15 FString ContentPath = FPaths::ProjectContentDir();
16 for (int32 ResidentIndex = 1; ResidentIndex <= 4; ++ResidentIndex)
17 {
18 FString ResidentFileName = FString::Printf(TEXT("Resident%dSettings.json"), ResidentIndex);
19 FString ResidentFilePath = FPaths::Combine(ContentPath, ResidentFileName);
20
21 // JSON 파일 내용을 저장할 문자열
22 FString ResidentJsonContent;
23 if (!FFileHelper::LoadFileToString(ResidentJsonContent, *ResidentFilePath))
24 {
25 // 파일 읽기 실패 시 로그 출력
26 UE_LOG(LogTemp, Error, TEXT("Failed to load Resident settings file at: %s"), *ResidentFilePath);
27 continue;
28 }
29
30 // JSON 문자열을 파싱하여 JSON 객체로 변환
31 TSharedPtr<FJsonObject> ResidentJson;
32 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(ResidentJsonContent);
33 if (FJsonSerializer::Deserialize(Reader, ResidentJson) && ResidentJson.IsValid())
34 {
35 // JSON에서 데이터 추출
36 FString Name = ResidentJson->GetStringField("Name");
37 FString Role = ResidentJson->GetStringField("Role");
38 FString Prompt = ResidentJson->GetStringField("Prompt");
39
40 // 로그로 데이터 확인
41 UE_LOG(LogTemp, Log, TEXT("Resident Name: %s"), *Name);
42 UE_LOG(LogTemp, Log, TEXT("Role: %s"), *Role);
43 UE_LOG(LogTemp, Log, TEXT("Prompt: %s"), *Prompt);
44
45 // 추출한 데이터를 기반으로 주민 초기화 로직 구현 가능
46 // 예: 특정 속성에 저장하거나 게임 상태에 따라 사용
47 }
48 else
49 {
50 // JSON 파싱 실패 시 오류 로그 출력
51 UE_LOG(LogTemp, Error, TEXT("Failed to parse Resident settings JSON file: %s"), *ResidentFilePath);
52 }
53 }
54}
