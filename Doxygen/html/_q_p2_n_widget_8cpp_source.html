38void UQP2NWidget::HandleEnterKeyPress()
39{
40 /*예외처리*/
41 //공백문자열이면 무시한다.
42 if (inputBox->GetText().ToString() == TEXT("")) {
43 UE_LOG(LogLogic, Log, TEXT("공백문자열을 input했습니다. 리턴합니다."));
44 return;
45 }
47 //1 .플레이어 텍스트를 inputbox에 있던거로 업뎃한다.
48 UpdatePlayerText(inputBox->GetText().ToString());
49 FString PlayerInput = inputBox->GetText().ToString();
50 //1-2. 서버에 플레이어 대사 저장
51 AQPlayerState* PlayerState = GetOwningLocalPlayer()->PlayerController->GetPlayerState<AQPlayerState>();
52 if (PlayerState == nullptr)
53 {
54 UE_LOG(LogTemp, Error, TEXT("UQP2NWidget::HandleEnterKeyPress - Can't find PlayerState."));
55 }
56 // 플레이어 응답 대화기록에 저장
57 TObjectPtr<UNPCComponent> NPCComponent = ConversingNPC->GetPawn()->FindComponentByClass<UNPCComponent>();
58 int32 NPCID = NPCComponent->GetNPCID();
59 PlayerState->ServerRPCAddP2NPlayerStatement(EConversationType::P2N, NPCID, PlayerState->GetPlayerId(), PlayerInput);
60
61 //2. inputbox는 공란으로 만든다.
62 inputBox->SetText(FText::FromString(TEXT("")));
63 //3. NPC Text는 음.. 으로 바꾼다.
64 UpdateNPCText(WhenGenerateResponseText);
65 //4. NPC에게 응답을 요청한다.
66 ConversingNPC->Response(PlayerInput,EConversationType::P2N);
67}
