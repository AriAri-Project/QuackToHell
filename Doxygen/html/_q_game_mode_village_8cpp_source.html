44void AQGameModeVillage::BeginPlay()
45{
46 Super::BeginPlay();
47
48 // NPC ID 카운터 리셋
49 UJuryComponent::JuryCount = 0;

51 UE_LOG(LogTemp, Log, TEXT("NPCID 카운터 초기화 완료 (JuryCount = 0, ResidentCount = 0)"));
52
53
54 UE_LOG(LogTemp, Log, TEXT("AAIGameModeTest::BeginPlay() 실행됨"));
55
56 UWorld* World = GetWorld();
57 if (!World)
58 {
59 UE_LOG(LogTemp, Error, TEXT("❌ BeginPlay - World is nullptr!"));
60 return;
61 }
62
63 // 기존 GodActor 찾기
64 TArray<AActor*> FoundActors;
65 UGameplayStatics::GetAllActorsWithTag(World, TEXT("GodActor"), FoundActors);
66
67 if (FoundActors.Num() == 0 || !FoundActors[0])
68 {
69 UE_LOG(LogTemp, Error, TEXT("❌ BeginPlay - GodActor를 찾을 수 없음!"));
70 return;
71 }
72
73 AActor* GodActor = FoundActors[0];
74 UGodCall* GodCall = GodActor->FindComponentByClass<UGodCall>();
75
76 if (!GodCall)
77 {
78 UE_LOG(LogTemp, Error, TEXT("❌ BeginPlay - GodCall을 찾을 수 없음!"));
79 return;
80 }
81
82 UE_LOG(LogTemp, Log, TEXT("✅ BeginPlay - 프롬프트 생성 완료 후 게임 진행"));
83}

103void AQGameModeVillage::TravelToCourtMap()
104{
105 if (!HasAuthority()) return;
106
107 // 모든 플레이어가 준비되지 않았다면 return
108 for (TActorIterator<AQPlayerState> IT(GetWorld()); IT; ++IT)
109 {
110 if (!IT->GetIsReadyToTravelToCourt())
111 {
112 UE_LOG(LogLogic, Log, TEXT("AQVillageGameState::ServerRPCRequestTravelToCourt_Implementation - %s is not ready for travel to court."), *IT->GetName());
113 return;
114 }
115 }
116
117 // 1) GameMode 에서 seamless travel 켜기
118 bUseSeamlessTravel = true;
119
120 // 2) 서버 주도 맵 전환 (listen 모드 유지)
121 FString URL = TEXT("/Game/Maps/NextMap?listen");
122 GetWorld()->ServerTravel(URL);
123}
