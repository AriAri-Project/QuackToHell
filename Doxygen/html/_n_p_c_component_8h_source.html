38USTRUCT()
39struct FOpenAIResponse
40{
41 GENERATED_BODY()
42
43 FString ResponseText;
44
45 static FOpenAIResponse FromJson(const FString& JsonContent)
46 {
47 FOpenAIResponse Response;
48 TSharedPtr<FJsonObject> JsonObject;
49 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(JsonContent);
50
51 if (FJsonSerializer::Deserialize(Reader, JsonObject) && JsonObject.IsValid())
52 {
53 const TArray<TSharedPtr<FJsonValue>>* ChoicesArray;
54 if (JsonObject->TryGetArrayField("choices", ChoicesArray) && ChoicesArray->Num() > 0)
55 {
56 Response.ResponseText = (*ChoicesArray)[0]->AsString();
57 }
58 }
59
60 if (Response.ResponseText.IsEmpty())
61 {
62 Response.ResponseText = TEXT("죄송합니다, 답변할 수 없습니다.");
63 }
64
65 return Response;
66 }
67};
