52USTRUCT()
53struct FOpenAIResponse
54{
55 GENERATED_BODY()
56
57 FString ResponseText;
58
59 static FOpenAIResponse FromJson(const FString& JsonContent)
60 {
61 FOpenAIResponse Response;
62 TSharedPtr<FJsonObject> JsonObject;
63 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(JsonContent);
64
65 if (FJsonSerializer::Deserialize(Reader, JsonObject) && JsonObject.IsValid())
66 {
67 const TArray<TSharedPtr<FJsonValue>>* ChoicesArray;
68 if (JsonObject->TryGetArrayField("choices", ChoicesArray) && ChoicesArray->Num() > 0)
69 {
70 Response.ResponseText = (*ChoicesArray)[0]->AsString();
71 }
72 }
73
74 if (Response.ResponseText.IsEmpty())
75 {
76 Response.ResponseText = TEXT("죄송합니다, 답변할 수 없습니다.");
77 }
78
79 return Response;
80 }
81};
