50 FString ToJson() const
51 {
52 TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject);
53 JsonObject->SetStringField("prompt", Prompt);
54 JsonObject->SetNumberField("max_tokens", MaxTokens);
55 JsonObject->SetNumberField("speaker_id", SpeakerID);
56 JsonObject->SetNumberField("listener_id", ListenerID);
57 JsonObject->SetStringField("conversation_type", ConversationTypeToString(ConversationType));
58
59 FString OutputString;
60 TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&OutputString);
61 FJsonSerializer::Serialize(JsonObject.ToSharedRef(), Writer);

72 case EConversationType::N2N: return TEXT("N2N");
73 case EConversationType::NMonologue: return TEXT("NMonologue");

86USTRUCT(BlueprintType)
87struct FOpenAIResponse
88{
89 GENERATED_BODY()
90
91 FString ResponseText;
92 int32 SpeakerID;
93 int32 ListenerID;
94 EConversationType ConversationType;
95
96 static FOpenAIResponse FromJson(const FString& JsonContent)
97{
98 FOpenAIResponse Response;
99 TSharedPtr<FJsonObject> JsonObject;
100 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(JsonContent);
101
102 if (!FJsonSerializer::Deserialize(Reader, JsonObject) || !JsonObject.IsValid())
103 {
104 UE_LOG(LogTemp, Error, TEXT("OpenAI 응답 JSON 파싱 실패!"));
105 return Response;
106 }
107
108 // ✅ "choices" 배열에서 첫 번째 "message.content" 값을 추출
109 const TArray<TSharedPtr<FJsonValue>>* ChoicesArray;
110 if (JsonObject->TryGetArrayField(TEXT("choices"), ChoicesArray) && ChoicesArray->Num() > 0)
111 {
112 TSharedPtr<FJsonObject> FirstChoice = (*ChoicesArray)[0]->AsObject();
113 if (FirstChoice.IsValid() && FirstChoice->HasField(TEXT("message")))
114 {
115 TSharedPtr<FJsonObject> MessageObj = FirstChoice->GetObjectField(TEXT("message"));
116 if (MessageObj.IsValid() && MessageObj->HasField(TEXT("content")))
117 {
118 Response.ResponseText = MessageObj->GetStringField(TEXT("content"));
119 }
120 }
121 }
122
123 // ✅ speaker_id, listener_id, conversation_type 필드가 존재할 경우만 할당
124 if (JsonObject->HasField("speaker_id"))
125 {
126 Response.SpeakerID = JsonObject->GetIntegerField("speaker_id");
127 }
128
129 if (JsonObject->HasField("listener_id"))
130 {
131 Response.ListenerID = JsonObject->GetIntegerField("listener_id");
132 }
133
134 FString ConversationTypeString;
135 if (JsonObject->TryGetStringField("conversation_type", ConversationTypeString))
136 {
137 if (ConversationTypeString == "P2N") Response.ConversationType = EConversationType::P2N;
138 else if (ConversationTypeString == "N2N") Response.ConversationType = EConversationType::N2N;
139 else if (ConversationTypeString == "NMonologue") Response.ConversationType = EConversationType::NMonologue;
140 else if (ConversationTypeString == "OpeningStatement") Response.ConversationType = EConversationType::OpeningStatement;
141 }
142
143 UE_LOG(LogTemp, Log, TEXT("OpenAI 응답 파싱 완료: %s"), *Response.ResponseText);
144 return Response;
145}
146};
