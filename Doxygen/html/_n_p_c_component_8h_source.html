95 int32 SpeakerID = -1;

98 int32 ListenerID = -1;

100 UPROPERTY(BlueprintReadWrite, Category = "AI")
101 EConversationType ConversationType = EConversationType::None;

103 static FOpenAIResponse FromJson(const FString& JsonContent)
104{
105 FOpenAIResponse Response;
106 TSharedPtr<FJsonObject> JsonObject;
107 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(JsonContent);
108
109 if (!FJsonSerializer::Deserialize(Reader, JsonObject) || !JsonObject.IsValid())
110 {
111 UE_LOG(LogTemp, Error, TEXT("OpenAI 응답 JSON 파싱 실패!"));
112 return Response;
113 }
114
115 // ✅ "choices" 배열에서 첫 번째 "message.content" 값을 추출
116 const TArray<TSharedPtr<FJsonValue>>* ChoicesArray;
117 if (JsonObject->TryGetArrayField(TEXT("choices"), ChoicesArray) && ChoicesArray->Num() > 0)
118 {
119 TSharedPtr<FJsonObject> FirstChoice = (*ChoicesArray)[0]->AsObject();
120 if (FirstChoice.IsValid() && FirstChoice->HasField(TEXT("message")))
121 {
122 TSharedPtr<FJsonObject> MessageObj = FirstChoice->GetObjectField(TEXT("message"));
123 if (MessageObj.IsValid() && MessageObj->HasField(TEXT("content")))
124 {
125 Response.ResponseText = MessageObj->GetStringField(TEXT("content"));
126 }
127 }
128 }
129
130 // ✅ speaker_id, listener_id, conversation_type 필드가 존재할 경우만 할당
131 if (JsonObject->HasField("speaker_id"))
132 {
133 Response.SpeakerID = JsonObject->GetIntegerField("speaker_id");
134 }
135
136 if (JsonObject->HasField("listener_id"))
137 {
138 Response.ListenerID = JsonObject->GetIntegerField("listener_id");
139 }
140
141 FString ConversationTypeString;
142 if (JsonObject->TryGetStringField("conversation_type", ConversationTypeString))
143 {
144 if (ConversationTypeString == "P2N") Response.ConversationType = EConversationType::P2N;
145 else if (ConversationTypeString == "N2N") Response.ConversationType = EConversationType::N2N;
146 else if (ConversationTypeString == "NMonologue") Response.ConversationType = EConversationType::NMonologue;
147 else if (ConversationTypeString == "OpeningStatement") Response.ConversationType = EConversationType::OpeningStatement;
148 }
149
150 UE_LOG(LogTemp, Log, TEXT("OpenAI 응답 파싱 완료: %s"), *Response.ResponseText);
151 return Response;
152}
153};
