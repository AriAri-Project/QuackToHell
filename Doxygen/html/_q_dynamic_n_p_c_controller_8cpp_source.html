64void AQDynamicNPCController::UpdateRotation()
65{
66
67 //현재 NPC 위치
68 FVector NPCPosition = GetPawn()->GetActorLocation();
69
70 //상대방의 위치
71 FVector OpponentPosition = OpponentPawn->GetActorLocation();
72
73 //목표 방향으로 회전 계산
74 FRotator LookAtRotation = UKismetMathLibrary::FindLookAtRotation(NPCPosition, OpponentPosition);
75
76 //현재 Rotation 가져오기
77 FRotator CurrentRotation = GetPawn()->GetActorRotation();
78
79 //부드러운 회전 적용 (보간)
80 FRotator NewRotation = FMath::RInterpTo(CurrentRotation, LookAtRotation, GetWorld()->GetDeltaSeconds(), 5.0f);
81 NewRotation = FRotator(CurrentRotation.Pitch, NewRotation.Yaw, CurrentRotation.Roll);
82
83 //회전 적용
84 GetPawn()->SetActorRotation(NewRotation);
85
86 if (FMath::Abs(CurrentRotation.Yaw - LookAtRotation.Yaw) < 1.0f) {
87 SetActorTickEnabled(false);
88 bIsRotating = false;
89 OpponentPawn = nullptr;
90 }
91}

106 }
107
108 UE_LOG(LogLogic, Log, TEXT("NPC 이동 재개."));

111void AQDynamicNPCController::FreezePawn()
112{
113 //폰 정보 가져오기
114 APawn* ControlledPawn = this->GetPawn();
115 if (!ControlledPawn) {
116 UE_LOG(LogLogic, Error, TEXT("FreezePawn: ControlledPawn is nullptr"));
117 return;
118 }
119
120 //movement component 가져오기
121 TObjectPtr<UCharacterMovementComponent> MovementComponent = ControlledPawn->FindComponentByClass<UCharacterMovementComponent>();
122 if (MovementComponent) {
123 MovementComponent->StopMovementImmediately();//즉시멈춤
124 MovementComponent->SetComponentTickEnabled(false);//틱 비활성화
125 MovementComponent->Deactivate();//이동 비활성화
126 }
127 else if (!MovementComponent) {
128 UE_LOG(LogLogic, Error, TEXT("FreezePawn: MovementComponent is nullptr"));
129 return;
130 }
131
132 UE_LOG(LogLogic, Log, TEXT("NPC 멈춤."));
133}
