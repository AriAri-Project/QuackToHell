48TObjectPtr<AActor> AQPlayer::GetClosestNPC()
49{
50 if (OverlappingNPCs.Num() == 0) {
51 //대화 대상 없음
52 return nullptr;
53 }
54
55 TObjectPtr<AActor> ClosestNPC = nullptr;
56 float MinDistance = FLT_MAX;
57
58 for (TObjectPtr<AActor> NPC : OverlappingNPCs) {
59 //플레이어와 NPC간 거리
60 float Distance = FVector::Dist(this->GetActorLocation(), NPC->GetActorLocation());
61 //최소거리찾기
62 if (Distance < MinDistance) {
63 MinDistance = Distance;
64 ClosestNPC = NPC;
65 }
66 }
67
68 return ClosestNPC;
69}
