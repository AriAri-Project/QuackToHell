78TObjectPtr<AActor> AQPlayer::GetClosestNPC()
79{
80 if (OverlappingNPCs.Num() == 0) {
81 //대화 대상 없음
82 return nullptr;
83 }
84
85 TObjectPtr<AActor> ClosestNPC = nullptr;
86 float MinDistance = FLT_MAX;
87
88 for (TObjectPtr<AActor> NPC : OverlappingNPCs) {
89 //플레이어와 NPC간 거리
90 float Distance = FVector::Dist(this->GetActorLocation(), NPC->GetActorLocation());
91 //최소거리찾기
92 if (Distance < MinDistance) {
93 MinDistance = Distance;
94 ClosestNPC = NPC;
95 }
96 }
97 return ClosestNPC;
98}

196void AQPlayer::ServerRPCCanStartConversP2N_Implementation(AQPlayerController* TargetController, AQNPC* NPC)
197{
198 bool bResult = true;
199 TObjectPtr<UNPCComponent> NPCComponent = NPC->FindComponentByClass<UNPCComponent>();
200
201 if (ThisPlayerState == nullptr || NPCComponent == nullptr)
202 {
203 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> LocalPlayerState or NPCComponent is null."));
204 bResult = false;
205 }
206
207 // OpenAI에 Request를 보낼 수 있는지 확인
208 if (!NPCComponent->CanSendOpenAIRequest())
209 {
210 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> CanSendOpenAIRequest is false"));
211 bResult = false;
212 }
213 // Player ConversationState가 None인지 확인
214 else if (ThisPlayerState->GetPlayerConversationState() != EConversationType::None)
215 {
216 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> Player ConversationState is not None but %s"), *UEnum::GetValueAsString(ThisPlayerState->GetPlayerConversationState()) );
217 bResult = false;
218 }
219 // NPC ConversationState가 None인지 확인
220 else if (NPC->GetNPCConversationState() != EConversationType::None)
221 {
222 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> NPC ConversationState is not None"));
223 bResult = false;
224 }
225 // NPC와 이전에 대화한 적 있는지 확인
226 else if (!NPCComponent->GetIsFirstConversation())
227 {
228 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> This is not a first conversation."));
229 bResult = false;
230 }
231 TargetController->ClientRPCUpdateCanStartConversP2N(bResult);
232}

262void AQPlayer::ServerRPCStartConversation_Implementation(AQPlayerController* ClientPC, AQNPC* NPC)
263{
264 // 상태 업데이트
265 ThisPlayerState->SetPlayerConverstationState(EConversationType::P2N);
266 NPC->SetNPCConversationState(EConversationType::P2N);
267
268 // NPC에게 대화 시작 알림
269 TObjectPtr<AQDynamicNPCController> DynamicNPCController = Cast<AQDynamicNPCController>(NPC->GetController());
270 if (DynamicNPCController == nullptr)
271 {
272 UE_LOG(LogLogic, Error, TEXT("AQPlayer::ServerRPCFinishConversation_Implementation - DynamicNPCController is nullptr."));
273 return;
274 }
275 DynamicNPCController->StartDialog(this, ENPCConversationType::P2N);
276
277 // 다른 플레이어들 시점 처리
278 for (TActorIterator<AQPlayer> It(GetWorld()); It; ++It)
279 {
280 AQPlayer* Peer = *It;
281 Peer->MulticastRPCStartConversation(this, NPC);
282 }
283
284 // OpenAI에게 NPC의 첫 대사 요청하기
285 FString Temp = TEXT("");
286 FOpenAIRequest Request(
287 NPC->FindComponentByClass<UNPCComponent>()->GetNPCID(),
288 ThisPlayerState->GetPlayerId(),
289 EConversationType::PStart,
290 Temp
291 );
292 NPC->FindComponentByClass<UNPCComponent>()->ServerRPCGetNPCResponse(ClientPC,Request);
293}

295void AQPlayer::ServerRPCFinishConversation_Implementation(AQPlayerController* TargetController, AQNPC* NPC)
296{
297 bool bResult = false;
298 // 상태 업데이트
299 if (ThisPlayerState == nullptr || NPC == nullptr)
300 {
301 return;
302 }
303 ThisPlayerState->SetPlayerConverstationState(EConversationType::None);
304 NPC->SetNPCConversationState(EConversationType::None);
305
306 // NPC에게 대화 종료 알림
307 TObjectPtr<AQDynamicNPCController> DynamicNPCController = Cast<AQDynamicNPCController>(NPC->GetController());
308 if (DynamicNPCController == nullptr)
309 {
310 UE_LOG(LogLogic, Error, TEXT("AQPlayer::ServerRPCFinishConversation_Implementation - DynamicNPCController is nullptr."));
311 return;
312 }
313 DynamicNPCController->EndDialog();
314
315 // 다른 플레이어들 시점 처리
316 for (TActorIterator<AQPlayer> It(GetWorld()); It; ++It)
317 {
318 AQPlayer* Peer = *It;
319 Peer->MulticastRPCFinishConversation(this, NPC);
320 }
321
322 //클라이언트에게 대화끝내기 처리 요청
323 TargetController->ClientRPCFinishConversation(NPC);
324}
