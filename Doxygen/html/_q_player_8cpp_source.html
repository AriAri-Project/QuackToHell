55TObjectPtr<AActor> AQPlayer::GetClosestNPC()
56{
57 if (OverlappingNPCs.Num() == 0) {
58 //대화 대상 없음
59 return nullptr;
60 }
61
62 TObjectPtr<AActor> ClosestNPC = nullptr;
63 float MinDistance = FLT_MAX;
64
65 for (TObjectPtr<AActor> NPC : OverlappingNPCs) {
66 //플레이어와 NPC간 거리
67 float Distance = FVector::Dist(this->GetActorLocation(), NPC->GetActorLocation());
68 //최소거리찾기
69 if (Distance < MinDistance) {
70 MinDistance = Distance;
71 ClosestNPC = NPC;
72 }
73 }
74
75 return ClosestNPC;
76}
