78TObjectPtr<AActor> AQPlayer::GetClosestNPC()
79{
80 if (OverlappingNPCs.Num() == 0) {
81 //대화 대상 없음
82 return nullptr;
83 }
84
85 TObjectPtr<AActor> ClosestNPC = nullptr;
86 float MinDistance = FLT_MAX;
87
88 for (TObjectPtr<AActor> NPC : OverlappingNPCs) {
89 //플레이어와 NPC간 거리
90 float Distance = FVector::Dist(this->GetActorLocation(), NPC->GetActorLocation());
91 //최소거리찾기
92 if (Distance < MinDistance) {
93 MinDistance = Distance;
94 ClosestNPC = NPC;
95 }
96 }
97
98 return ClosestNPC;
99}

159void AQPlayer::ServerRPCCanStartConversP2N_Implementation(AQPlayerController* TargetController, AQNPC* NPC)
160{
161 bool bResult = true;
162 TObjectPtr<UNPCComponent> NPCComponent = NPC->FindComponentByClass<UNPCComponent>();
163
164 if (LocalPlayerState == nullptr || NPCComponent == nullptr)
165 {
166 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> LocalPlayerState or NPCComponent is null."));
167 bResult = false;
168 }
169
170 // OpenAI에 Request를 보낼 수 있는지 확인
171 if (!NPCComponent->CanSendOpenAIRequest())
172 {
173 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> CanSendOpenAIRequest is false"));
174 bResult = false;
175 }
176 // Player ConversationState가 None인지 확인
177 else if (LocalPlayerState->GetPlayerConversationState() != EConversationType::None)
178 {
179 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> Player ConversationState is not None but %s"), *UEnum::GetValueAsString(LocalPlayerState->GetPlayerConversationState()) );
180 bResult = false;
181 }
182 // NPC ConversationState가 None인지 확인
183 else if (NPC->GetNPCConversationState() != EConversationType::None)
184 {
185 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> NPC ConversationState is not None"));
186 bResult = false;
187 }
188 // NPC와 이전에 대화한 적 있는지 확인
189 else if (!NPCComponent->GetIsFirstConversation())
190 {
191 UE_LOG(LogLogic, Log, TEXT("ServerRPCCanStartConversP2N_Implementation -> This is not a first conversation."));
192 bResult = false;
193 }
194 TargetController->ClientRPCUpdateCanStartConversP2N(bResult);
195}

225void AQPlayer::ServerRPCStartConversation_Implementation(AQNPC* NPC)
226{
227 // 상태 업데이트
228 LocalPlayerState->SetPlayerConverstationState(EConversationType::P2N);
229 NPC->SetNPCConversationState(EConversationType::P2N);
230
231 // 다른 플레이어들 시점 처리
232 AQPlayer* LocalPlayer = Cast<AQPlayer>(LocalPlayerState->GetPawn());
233 if (LocalPlayer)
234 {
235 MulticastRPCStartConversation(LocalPlayer, NPC);
236 }
237
238 // OpenAI에게 NPC의 첫 대사 요청하기
239 FString Temp = TEXT("");
240 FOpenAIRequest Request(
241 LocalPlayerState->GetPlayerId(),
242 NPC->FindComponentByClass<UNPCComponent>()->GetNPCID(),
243 EConversationType::PStart,
244 Temp
245 );
246 NPC->FindComponentByClass<UNPCComponent>()->ServerRPCGetNPCResponse(Request);
247}
