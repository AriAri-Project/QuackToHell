95TObjectPtr<AQVillageUIManager> AQVillageUIManager::GetInstance(TObjectPtr<UWorld> World)
96{
97 ENetMode NetMode = World->GetNetMode();
98 if (NetMode == NM_Client) {
99 UE_LOG(LogLogic, Log, TEXT("클라이언트!"));
100 }
101 if (!Instance)
102 {
103 if (NetMode == NM_DedicatedServer || NetMode == NM_ListenServer)
104 {
105 // 서버에서 Instance 생성
106 UE_LOG(LogLogic, Log, TEXT("서버에서 AQVillageUIManager 생성"));
107 Instance = World->SpawnActor<AQVillageUIManager>(AQVillageUIManager::StaticClass());
108
109 }
110 else if (NetMode == NM_Client)
111 {
112 // 클라이언트에서 Instance 생성
113 UE_LOG(LogLogic, Log, TEXT("클라이언트에서 AQVillageUIManager 생성"));
114 Instance = World->SpawnActor<AQVillageUIManager>(AQVillageUIManager::StaticClass());
115
116 }
117 }
118
119 return Instance;

219void AQVillageUIManager::TurnOnUI(EVillageUIType UIType)
220{
221 /*예외처리*/
222 //키에 매핑되는 클래스목록이 존재하지 않으면
223 if (!UIWidgetClasses.Contains(UIType)) {
224 UE_LOG(LogLogic, Warning, TEXT("QVillageUIManager: 키에 매핑되는 클래스목록이 존재하지 않습니다."));
225 return;
226 }
227
228 /*UI켜기*/
229 //위젯 이미 만들어져있으면 visible 전환
230 if (ActivedVillageWidgets.Contains(UIType)) {
231 ActivedVillageWidgets[UIType]->SetVisibility(ESlateVisibility::Visible);
232 return;
233 }
234
235 //없으면 위젯 새로 만들기 & visible처리
236 TObjectPtr<UWorld> World = GetWorld();
237 if (!World) return;
238
239 TObjectPtr<UUserWidget> NewWidget = CreateWidget<UUserWidget>(World, UIWidgetClasses[UIType]);
240 if (NewWidget) {
241 NewWidget->AddToViewport();
242 NewWidget->SetVisibility(ESlateVisibility::Visible);
243 ActivedVillageWidgets.Add(UIType, NewWidget);
244 }
245}
