34void AQCharacter::BeginPlay()
35{
36 Super::BeginPlay();
37
38 if (NameWidgetComponent)
39 {
40 UUserWidget* UserWidget = NameWidgetComponent->GetWidget();
41 if (UserWidget)
42 {
43 NameWidget = Cast<UQNameWidget>(UserWidget);
44 }
45 }
46
47 // 서버에서 실행 -> 오너십을 부여하기 위해
48 if (HasAuthority())
49 {
50 APlayerController* LocalPlayerController = nullptr;
51
52 // GetWorld()에서 모든 플레이어 컨트롤러를 순회하며, 로컬 플레이어 컨트롤러 찾기
53 for (FConstPlayerControllerIterator It = GetWorld()->GetPlayerControllerIterator(); It; ++It)
54 {
55 APlayerController* PC = Cast<APlayerController>(*It);
56 if (PC && PC->IsLocalPlayerController()) // 로컬 플레이어인지 확인
57 {
58 LocalPlayerController = PC;
59 break;
60 }
61 }
62
63 if (LocalPlayerController)
64 {
65 SetOwner(LocalPlayerController);
66 UE_LOG(LogLogic, Log, TEXT("%s 's Owner is %s"), *this->GetName(), *GetOwner()->GetName());
67 }
68 else
69 {
70 UE_LOG(LogTemp, Error, TEXT("Failed to find Local Player Controller!"));
71 }
72 }
