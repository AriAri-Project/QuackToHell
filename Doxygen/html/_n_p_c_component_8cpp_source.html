63// NPC의 프롬프트 파일 로드
64bool UNPCComponent::LoadPrompt()
65{
66 if (PromptFilePath.IsEmpty())
67 {
68 UE_LOG(LogTemp, Warning, TEXT("LoadPrompt 실행 전 PromptFilePath가 비어 있음! 기본 경로 설정 시도."));
69
70 FString PromptFileName;
71
72 // NPC 역할(role)에 따라 JSON 파일명 결정
73 if (NPCRole == "resident")
74 {
75 PromptFileName = FString::Printf(TEXT("PromptToResident%d.json"), FCString::Atoi(*NPCID) - 2003);
76 }
77 else if (NPCRole == "jury")
78 {
79 PromptFileName = FString::Printf(TEXT("PromptToJury%d.json"), FCString::Atoi(*NPCID) - 2000);
80 }
81 else if (NPCRole == "defendant")
82 {
83 PromptFileName = TEXT("PromptToDefendant.json"); // Defendant는 단 하나만 존재
84 }
85 else
86 {
87 UE_LOG(LogTemp, Error, TEXT("LoadPrompt - 알 수 없는 NPC Role: %s"), *NPCRole);
88 return false;
89 }
90
91 // 파일 경로 설정
92 PromptFilePath = FPaths::ConvertRelativePathToFull(FPaths::Combine(FPaths::ProjectSavedDir(), TEXT("Prompt"), PromptFileName));
93
94 UE_LOG(LogTemp, Log, TEXT("LoadPrompt - NPCID: %s, Role: %s, 파일 경로: %s"), *NPCID, *NPCRole, *PromptFilePath);
95 }
96
97 FString FileContent;
98
99 if (!FFileHelper::LoadFileToString(FileContent, *PromptFilePath) || FileContent.IsEmpty())
100 {
101 UE_LOG(LogTemp, Error, TEXT("LoadPrompt 실패 (파일이 비어 있음) - %s"), *PromptFilePath);
102 return false;
103 }
104
105 TSharedPtr<FJsonObject> JsonObject;
106 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(FileContent);
107
108 if (!FJsonSerializer::Deserialize(Reader, JsonObject) || !JsonObject.IsValid())
109 {
110 UE_LOG(LogTemp, Error, TEXT("JSON 파싱 실패 - %s"), *PromptFilePath);
111 return false;
112 }
113
114 PromptContent = FileContent;
115
116 // 필수 필드 확인
117 if (JsonObject->HasField("npcid"))
118 {
119 NPCID = FString::FromInt(JsonObject->GetIntegerField("npcid"));
120 UE_LOG(LogTemp, Log, TEXT("LoadPrompt - NPCID 할당 완료: %s"), *NPCID);
121 }
122
123 if (JsonObject->HasField("name"))
124 {
125 NPCName = JsonObject->GetStringField("name");
126 UE_LOG(LogTemp, Log, TEXT("LoadPrompt - NPCName 할당 완료: %s"), *NPCName);
127 }
128
129 if (JsonObject->HasField("role"))
130 {
131 NPCRole = JsonObject->GetStringField("role");
132 UE_LOG(LogTemp, Log, TEXT("LoadPrompt - NPCRole 할당 완료: %s"), *NPCRole);
133 }
134
135 UE_LOG(LogTemp, Log, TEXT("LoadPrompt 완료 - NPCID=%s, Name=%s, Role=%s"), *NPCID, *NPCName, *NPCRole);
136 UE_LOG(LogTemp, Log, TEXT("PromptContent 저장 완료 - 내용: %s"), *PromptContent.Left(200));
137
138 return true;
139}

224FString UNPCComponent::ConvertJsonToReadableText(const FString& JsonString)
225{
226 TSharedPtr<FJsonObject> JsonObject;
227 TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(JsonString);
228
229 if (!FJsonSerializer::Deserialize(Reader, JsonObject) || !JsonObject.IsValid())
230 {
231 UE_LOG(LogTemp, Error, TEXT("🚨 JSON 파싱 실패! 원본 데이터 반환"));
232 return JsonString;
233 }
234
235 FString ReadableText;
236
237 // NPC 기본 정보
238 if (JsonObject->HasField("name"))
239 {
240 ReadableText += FString::Printf(TEXT("이름: %s\n"), *JsonObject->GetStringField("name"));
241 }
242
243 if (JsonObject->HasField("npcid"))
244 {
245 ReadableText += FString::Printf(TEXT("NPC ID: %d\n"), JsonObject->GetIntegerField("npcid"));
246 }
247
248 if (JsonObject->HasField("personality"))
249 {
250 ReadableText += FString::Printf(TEXT("성격: %s\n"), *JsonObject->GetStringField("personality"));
251 }
252
253 if (JsonObject->HasField("speech_style"))
254 {
255 ReadableText += FString::Printf(TEXT("말투: %s\n"), *JsonObject->GetStringField("speech_style"));
256 }
257
258 if (JsonObject->HasField("situation_understanding"))
259 {
260 ReadableText += FString::Printf(TEXT("현재 상황 인식: %s\n"), *JsonObject->GetStringField("situation_understanding"));
261 }
262
263 if (JsonObject->HasField("past_life_relevance"))
264 {
265 ReadableText += FString::Printf(TEXT("과거 관련성: %s\n"), *JsonObject->GetStringField("past_life_relevance"));
266 }
267
268 if (JsonObject->HasField("past_life_story"))
269 {
270 ReadableText += FString::Printf(TEXT("과거 이야기: %s\n"), *JsonObject->GetStringField("past_life_story"));
271 }
272
273 if (JsonObject->HasField("knowledge_of_defendant's_past_life"))
274 {
275 ReadableText += FString::Printf(TEXT("피고인의 과거에 대한 지식: %s\n"), *JsonObject->GetStringField("knowledge_of_defendant's_past_life"));
276 }
277
278 return ReadableText;
279}

282// P2N 대화 시작
283void UNPCComponent::StartConversation(FOpenAIRequest Request)
284{
285 UE_LOG(LogTemp, Log, TEXT("NPCComponent::StartConversation 실행 - NPCID: %s"), *NPCID);
286 UE_LOG(LogTemp, Log, TEXT("StartConversation 실행됨 - 현재 PromptContent 길이: %d"), PromptContent.Len());
287
288 Request.SpeakerID = FCString::Atoi(*GetPlayerIDAsString());
289 Request.ListenerID = GetNPCID();
290
291 if (PromptContent.IsEmpty())
292 {
293 UE_LOG(LogTemp, Error, TEXT("🚨 PromptContent가 비어 있음! NPCID: %s"), *NPCID);
294 return;
295 }
296
297 UE_LOG(LogTemp, Log, TEXT("Player started conversation with NPC %d: %s"), Request.ListenerID, *Request.Prompt);
298
299 FString ListenerNPCID = FString::FromInt(Request.ListenerID);
300 bool bIsFirstGreeting = !P2NDialogueHistory.Contains(ListenerNPCID) ||
301 P2NDialogueHistory[ListenerNPCID].DialogueLines.Num() == 0;
302
303 FOpenAIRequest AIRequest;
304 AIRequest.SpeakerID = Request.SpeakerID;
305 AIRequest.ListenerID = Request.ListenerID;
306 AIRequest.MaxTokens = 150;
307
308 // OpenAI가 JSON을 올바르게 해석할 수 있도록 변환
309 FString ReadablePromptContent = ConvertJsonToReadableText(PromptContent);
310
311 if (bIsFirstGreeting && Request.Prompt.IsEmpty())
312 {
313 AIRequest.ConversationType = EConversationType::PStart;
314 AIRequest.Prompt = FString::Printf(TEXT(
315 "{ \"model\": \"gpt-4o\", \"messages\": ["
316 "{ \"role\": \"system\", \"content\": \"당신은 마을 NPC입니다. 다음은 NPC의 설정입니다: %s\" },"
317 "{ \"role\": \"user\", \"content\": \"플레이어가 NPC를 처음 만났을 때 당신이 할 인사는?\" }],"
318 "\"max_tokens\": 150 }"
319 ), *ReadablePromptContent);
320 }
321 else
322 {
323 AIRequest.ConversationType = EConversationType::P2N;
324 FString EscapedPlayerInput = Request.Prompt.Replace(TEXT("\n"), TEXT(" ")).Replace(TEXT("\""), TEXT("'"));
325 AIRequest.Prompt = FString::Printf(TEXT(
326 "{ \"model\": \"gpt-4o\", \"messages\": ["
327 "{ \"role\": \"system\", \"content\": \"당신은 마을 NPC입니다. 다음은 NPC의 설정입니다: %s\" },"
328 "{ \"role\": \"user\", \"content\": \"플레이어의 질문: '%s'\" }],"
329 "\"max_tokens\": 150 }"
330 ), *ReadablePromptContent, *EscapedPlayerInput);
331 }
332
333 UE_LOG(LogTemp, Log, TEXT("OpenAI 최종 요청 데이터(JSON): %s"), *AIRequest.Prompt);
334
335 RequestOpenAIResponse(AIRequest, [this, Request](FOpenAIResponse AIResponse)
336 {
337 if (AIResponse.ResponseText.IsEmpty())
338 {
339 AIResponse.ResponseText = TEXT("죄송합니다, 질문에 답할 수 없습니다.");
340 }
341
342 ResponseCache.Add(Request.Prompt, AIResponse.ResponseText);
343 SendNPCResponseToServer(AIResponse);
344 SaveP2NDialogue(Request, AIResponse);
345 });
346}

349// N2N 대화 시작
350void UNPCComponent::StartNPCToNPCDialog(const FOpenAIRequest& Request)
351{
352 FString SpeakerNPCID = FString::FromInt(Request.SpeakerID);
353 FString ListenerNPCID = FString::FromInt(Request.ListenerID);
354
355 UE_LOG(LogTemp, Log, TEXT("StartNPCToNPCDialog 시작: %s → %s"), *SpeakerNPCID, *ListenerNPCID);
356
357 FString PlayerDialogueSummary = TEXT("(플레이어와의 대화 기록 없음)");
358
359 if (P2NDialogueHistory.Contains(SpeakerNPCID) && P2NDialogueHistory[SpeakerNPCID].DialogueLines.Num() > 0)
360 {
361 TArray<FString>& DialogueLines = P2NDialogueHistory[SpeakerNPCID].DialogueLines;
362 int32 NumLines = DialogueLines.Num();
363
364 FString RecentLines;
365 int32 StartIndex = FMath::Max(0, NumLines - 3);
366 for (int32 i = StartIndex; i < NumLines; i++)
367 {
368 RecentLines += DialogueLines[i] + TEXT(" ");
369 }
370
371 PlayerDialogueSummary = RecentLines.TrimEnd();
372 }
373
374 FString ReadablePromptContent = ConvertJsonToReadableText(PromptContent);
375 FOpenAIRequest AIRequest;
376 AIRequest.Prompt = FString::Printf(TEXT(
377 "NPC '%s'가 NPC '%s'에게 말을 겁니다.\n"
378 "최근 플레이어와 나눈 대화 요약: %s\n"
379 "NPC '%s'의 성격을 반영해 첫 대화를 생성하세요."),
380 *SpeakerNPCID, *ListenerNPCID, *PlayerDialogueSummary, *SpeakerNPCID);
381
382 AIRequest.MaxTokens = 150;
383 AIRequest.SpeakerID = Request.SpeakerID;
384 AIRequest.ListenerID = Request.ListenerID;
385 AIRequest.ConversationType = EConversationType::N2N;
386
387 RequestOpenAIResponse(AIRequest, [this, ListenerNPCID, SpeakerNPCID](FOpenAIResponse AIResponse)
388 {
389 SendNPCResponseToServer(AIResponse);
390 ContinueNPCToNPCDialog(FOpenAIRequest(AIResponse.ListenerID, AIResponse.SpeakerID, EConversationType::N2N, AIResponse.ResponseText, 4));
391 });
392
393 UE_LOG(LogTemp, Log, TEXT("NPC-to-NPC 대화 요청 전송 완료"));
394}

397// N2N 대화 이어나감
398void UNPCComponent::ContinueNPCToNPCDialog(const FOpenAIRequest& Request)
399{
400 if (Request.MaxTokens <= 0)
401 {
402 UE_LOG(LogTemp, Log, TEXT("N2N 대화가 종료되었습니다."));
403 return;
404 }
405
406 FString SpeakerNPCID = FString::FromInt(Request.SpeakerID);
407 FString ListenerNPCID = FString::FromInt(Request.ListenerID);
408
409 UE_LOG(LogTemp, Log, TEXT("ContinueNPCToNPCDialog: %s → %s"), *SpeakerNPCID, *ListenerNPCID);
410
411 // 최근 대화 내역을 반영하여 응답 생성
412 FString PreviousResponse = Request.Prompt;
413 FString ReadablePromptContent = ConvertJsonToReadableText(PromptContent);
414
415 FOpenAIRequest AIRequest;
416 AIRequest.Prompt = FString::Printf(TEXT(
417 "NPC '%s'가 NPC '%s'의 말에 대답합니다.\n"
418 "이전 대화: %s\n"
419 "자연스럽고 일관성 있는 대화를 이어가세요."),
420 *SpeakerNPCID, *ListenerNPCID, *PreviousResponse);
421
422 AIRequest.MaxTokens = Request.MaxTokens - 1;
423 AIRequest.SpeakerID = Request.SpeakerID;
424 AIRequest.ListenerID = Request.ListenerID;
425 AIRequest.ConversationType = EConversationType::N2N;
426
427
428 // OpenAI API 호출 후, 대화 이어나가기 (남은 턴 수 감소)
429 RequestOpenAIResponse(AIRequest, [this, Request](FOpenAIResponse AIResponse)
430 {
431 SendNPCResponseToServer(AIResponse);
432 FOpenAIRequest NextRequest = Request;
433 NextRequest.Prompt = AIResponse.ResponseText; // AI 응답을 다음 메시지로 설정
434 NextRequest.MaxTokens -= 1; // 남은 턴 감소
435
436 ContinueNPCToNPCDialog(NextRequest);
437 });
438}

482void UNPCComponent::TrialStatement(FOpenAIRequest Request)
483{
484 UE_LOG(LogTemp, Log, TEXT("재판 관련 AI 응답 요청 처리 시작 (Type: %d)"), static_cast<int32>(Request.ConversationType));
485
486 if (PromptContent.IsEmpty())
487 {
488 UE_LOG(LogTemp, Error, TEXT("PromptContent가 비어 있음! NPCID: %s"), *NPCID);
489 return;
490 }
491
492 // 기본 설정 (Speaker: NPC ID, Listener: 플레이어 ID)
493 Request.SpeakerID = GetNPCID();
494 Request.ListenerID = FCString::Atoi(*GetPlayerIDAsString());
495
496 FString ReadablePromptContent = UNPCComponent::ConvertJsonToReadableText(PromptContent);
497 FString EscapedPlayerInput = Request.Prompt;
498
499 // JSON 객체 생성
500 TSharedPtr<FJsonObject> RootObject = MakeShareable(new FJsonObject());
501 RootObject->SetStringField("model", "gpt-4o");
502
503 // 메시지 배열 미리 선언
504 TArray<TSharedPtr<FJsonValue>> Messages;
505 TSharedPtr<FJsonObject> SystemMessage = MakeShareable(new FJsonObject());
506 SystemMessage->SetStringField("role", "system");
507
508 switch (Request.ConversationType)
509 {
510 case EConversationType::OpeningStatement:
511 {
512 Request.SpeakerID = GetNPCID(); // 피고인은 2000
513 Request.ListenerID = FCString::Atoi(*GetPlayerIDAsString());
514
515 SystemMessage->SetStringField("content",
516 FString::Printf(TEXT("현재 당신은 재판장에 앉아 당신의 판결을 기다리고 있는 입장입니다. 당신에게 재판이 유리해지도록, 당신의 설정을 반영하여 진술하세요: %s"),
517 *ReadablePromptContent));
518 Messages.Add(MakeShareable(new FJsonValueObject(SystemMessage)));
519 }
520 break;
521
522 case EConversationType::EvidenceReview:
523 {
524 SystemMessage->SetStringField("content",
525 FString::Printf(TEXT("당신은 천국과 지옥을 결정하는 재판의 판사입니다. 다음은 당신의 설정입니다: %s"),
526 *ReadablePromptContent));
527 Messages.Add(MakeShareable(new FJsonValueObject(SystemMessage)));
528
529 TSharedPtr<FJsonObject> UserMessage = MakeShareable(new FJsonObject());
530 UserMessage->SetStringField("role", "user");
531 UserMessage->SetStringField("content",
532 FString::Printf(TEXT("다음은 검사와 변호사가 나눈 대화입니다:\n%s\n"
533 "이 대화를 보고 판사의 입장에서 요약해주세요.\n"
534 "형식: '검사측의 주장은 ~이고, 변호사측의 주장은 ~이었습니다.'"), *EscapedPlayerInput));
535 Messages.Add(MakeShareable(new FJsonValueObject(UserMessage)));
536 }
537 break;
538
539 case EConversationType::DefendantInterrogation:
540 {
541 SystemMessage->SetStringField("content",
542 FString::Printf(TEXT("당신은 재판을 받고 있는 피고인입니다. 다음은 당신의 전생 정보입니다:\n%s"),
543 *ReadablePromptContent));
544 Messages.Add(MakeShareable(new FJsonValueObject(SystemMessage)));
545
546 TSharedPtr<FJsonObject> UserMessage = MakeShareable(new FJsonObject());
547 UserMessage->SetStringField("role", "user");
548 UserMessage->SetStringField("content",
549 FString::Printf(TEXT("검사 혹은 변호사의 질문: '%s'\n당신의 성격과 과거에 맞는 대답을 해주세요."), *EscapedPlayerInput));
550 Messages.Add(MakeShareable(new FJsonValueObject(UserMessage)));
551 }
552 break;
553 case EConversationType::JuryFinalOpinion:

611void UNPCComponent::RequestOpenAIResponse(const FOpenAIRequest& AIRequest, TFunction<void(FOpenAIResponse)> Callback)
612{
613 if (!CanSendOpenAIRequest())
614 {
615 UE_LOG(LogTemp, Warning, TEXT("OpenAI가 호출 중이므로 새로운 요청을 보낼 수 없습니다."));
616 return;
617 }
618
619 bIsRequestInProgress = true;
620
621 FString ApiKey = GetAPIKey();
622 if (ApiKey.IsEmpty())
623 {
624 UE_LOG(LogTemp, Error, TEXT("OpenAI API Key가 없습니다!"));
625 bIsRequestInProgress = false;
626 return;
627 }
628
629 FString RequestBody = AIRequest.Prompt;
630 UE_LOG(LogTemp, Log, TEXT("OpenAI에 보낼 최종 JSON 데이터: %s"), *RequestBody);
631
632 TSharedPtr<IHttpRequest> Request = FHttpModule::Get().CreateRequest();
633 Request->SetURL("https://api.openai.com/v1/chat/completions");
634 Request->SetVerb("POST");
635 Request->SetHeader("Authorization", "Bearer " + ApiKey);
636 Request->SetHeader("Content-Type", "application/json");
637 Request->SetContentAsString(RequestBody);
638
639 Request->OnProcessRequestComplete().BindLambda([this, Callback](FHttpRequestPtr Request, FHttpResponsePtr Response, bool bWasSuccessful)
640 {
641 bIsRequestInProgress = false;
642
643 if (!bWasSuccessful || !Response.IsValid())
644 {
645 UE_LOG(LogTemp, Error, TEXT("OpenAI 요청 자체가 실패함!"));
646
647 FOpenAIResponse FailedResponse;
648 FailedResponse.ResponseText = TEXT("죄송합니다, 현재 답변할 수 없습니다.");
649 Callback(FailedResponse);
650 return;
651 }
652
653 int32 ResponseCode = Response->GetResponseCode();
654 FString ResponseContent = Response->GetContentAsString();
655
656 UE_LOG(LogTemp, Log, TEXT("OpenAI 응답 코드: %d"), ResponseCode);
657 UE_LOG(LogTemp, Log, TEXT("OpenAI 응답 본문: %s"), *ResponseContent);
658
659 if (ResponseCode != 200)
660 {
661 UE_LOG(LogTemp, Error, TEXT("OpenAI 응답 실패! HTTP 응답 코드: %d"), ResponseCode);
662 UE_LOG(LogTemp, Error, TEXT("OpenAI 오류 메시지: %s"), *ResponseContent);
663
664 FOpenAIResponse FailedResponse;
665 FailedResponse.ResponseText = TEXT("죄송합니다, 현재 답변할 수 없습니다.");
666 Callback(FailedResponse);
667 return;
668 }
669
670 // 정상 응답 처리
671 FOpenAIResponse AIResponse = FOpenAIResponse::FromJson(ResponseContent);
672
673 if (AIResponse.ResponseText.IsEmpty())
674 {
675 UE_LOG(LogTemp, Error, TEXT("OpenAI 응답이 비어 있음! 기본 응답 반환."));
676 AIResponse.ResponseText = TEXT("죄송합니다, 질문에 답할 수 없습니다.");
677 }
678
679 Callback(AIResponse);
680 });
681
682 Request->ProcessRequest();
683}
